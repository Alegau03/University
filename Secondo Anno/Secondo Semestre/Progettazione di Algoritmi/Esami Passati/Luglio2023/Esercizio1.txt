Def DFS(G,Vis,y):
    Vis[y]=1
    Stack S;
    S.push(y)


    While S!=0:
        y=S.top()

        while y.adiacenti()!=0:
            z=y.adiacenti[0]
            y.adiacenti.remove(0)

            if Vis[z]==-1:
                Vis[z]=1
                S.push(z)


        if y==S.top():
            S.pop()
            break
