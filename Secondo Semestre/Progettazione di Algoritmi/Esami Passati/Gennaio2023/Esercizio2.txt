

Def ElementoNonDoppione(A):
    n= A.lenght()

    if n==1:
        return A[0]
    if n==3:
        if A[0]==A[1]:
            return A[2]
        else if A[0]==A[2]:
            return A[1]
        else: 
            return A[0]
    
    m= n/2

    if A[m]==A[m-1]:
        m-=1
    
    if A[0:m]%2==1:
        ElementoNonDoppione(A[0:m])
    else:
        ElementoNonDoppione(A[m:n])
