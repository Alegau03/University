	S={}

	for v in V(G):
		for u in v.adiacenti():
			if v.colore()==u.colore():
				S.append((v,u))


	R= E(G)-S
	Sol={}
	For (u,v) in S:
		if (Sol + (u,v)).findCycle()==Null:
			Sol.append((u,v))

		if (Sol + (u,v)).findCycle()==Null:
			Sol.append((u,v))

	return Sol
